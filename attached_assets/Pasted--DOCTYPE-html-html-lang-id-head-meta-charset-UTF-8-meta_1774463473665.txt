function getFilteredData() {
  const q = (document.getElementById('search')?.value || '').toLowerCase();
  return DATA.filter(r => {
    if (state.am && !state.am.includes(r.am)) return false;
    if (state.status && !state.status.includes(r.phase)) return false;
    if (state.kontrak && !state.kontrak.includes(r.kontrak)) return false;
    if (q && !r.proyek.toLowerCase().includes(q) &&
             !r.pelanggan.toLowerCase().includes(q) &&
             !r.lopid.toLowerCase().includes(q)) return false;
    return true;
  });
}

// ============================================================
// FORMAT
// ============================================================
function fmtNilai(n) {
  if (!n && n !== 0) return '–';
  if (n >= 1e9) return `Rp ${(n/1e9).toFixed(2)}M`;
  if (n >= 1e6) return `Rp ${Math.round(n/1e6)} jt`;
  return `Rp ${n.toLocaleString('id-ID')}`;
}
function fmtNilaiShort(n) {
  if (n >= 1e9) return `${(n/1e9).toFixed(1)}M`;
  if (n >= 1e6) return `${Math.round(n/1e6)} jt`;
  return `${n}`;
}

function phaseClass(p) {
  if (p==='F0'||p==='F1') return 'ph-f0';
  if (p==='F2'||p==='F3') return 'ph-f2';
  return 'ph-f4';
}
function statusClass(p) {
  if (p==='F0'||p==='F1') return 's-f0';
  if (p==='F2'||p==='F3') return 's-f2';
  if (p==='F4'||p==='F5') return 's-f4';
  if (p==='WON') return 's-won';
  if (p==='LOST') return 's-lost';
  return 's-prospect';
}

function renderOverview(data) {
  // Bar chart
  const phases = ['F0','F1','F2','F3','F4','F5'];
  const phaseColors = {
    F0:'#93c5fd', F1:'#93c5fd',
    F2:'#378ADD', F3:'#378ADD',
    F4:'#5b21b6', F5:'#3730a3'
  };
  const counts = {};
  phases.forEach(p => counts[p] = data.filter(r=>r.phase===p).length);
  const maxCount = Math.max(...Object.values(counts), 1);

  document.getElementById('bar-chart-area').innerHTML = phases.map(p => `
    <div class="bar-row">
      <span class="bar-phase" style="color:${phaseColors[p]}">${p}</span>
      <div class="bar-track">
        <div class="bar-fill" style="background:${phaseColors[p]};width:${(counts[p]/maxCount*100).toFixed(0)}%"></div>
      </div>
      <span class="bar-count">${counts[p]}</span>
    </div>
  `).join('');

  // Progress
  const totalReal = data.reduce((s,r)=>s+r.nilai,0);
  const target = TARGETS[state.scope];
  const pct = target ? (totalReal / target * 100) : 0;
  const shortage = totalReal - target;

  document.getElementById('prog-title').textContent = `Capaian real vs target (${state.scope})`;
  document.getElementById('lbl-target').textContent = `Target ${state.scope}`;
  document.getElementById('prog-pct').textContent = pct.toFixed(1) + '%';
  document.getElementById('prog-bar').style.width = Math.min(pct, 100) + '%';
  document.getElementById('val-real').textContent = fmtNilai(totalReal);
  document.getElementById('val-target').textContent = fmtNilai(target);

  const sl = document.getElementById('shortage-line');
  if (shortage >= 0) {
    sl.textContent = `Surplus: +${fmtNilai(shortage)}`;
    sl.style.color = 'var(--green)';
  } else {
    sl.textContent = `Shortage: ${fmtNilai(shortage)}`;
    sl.style.color = 'var(--red)';
  }

  // KPI
  const activeAM = [...new Set(data.map(r=>r.am))].length;
  const won = data.filter(r=>r.phase==='F5').length;
  const avgNilai = data.length ? totalReal / data.length : 0;

  document.getElementById('kpi-grid').innerHTML = `
    <div class="kpi-item">
      <div class="kpi-lbl">Total LOP</div>
      <div class="kpi-val c-blue">${data.length}</div>
      <div class="kpi-sub">aktif</div>
    </div>
    <div class="kpi-item">
      <div class="kpi-lbl">Active AM</div>
      <div class="kpi-val">${activeAM}</div>
      <div class="kpi-sub">dari 7 AM</div>
    </div>
    <div class="kpi-item">
      <div class="kpi-lbl">Avg nilai / LOP</div>
      <div class="kpi-val c-amber" style="font-size:16px">${fmtNilaiShort(avgNilai)}</div>
    </div>
    <div class="kpi-item">
      <div class="kpi-lbl">Won (F5)</div>
      <div class="kpi-val c-green">${won}</div>
      <div class="kpi-sub">LOP closed</div>
    </div>
  `;
}

            <td class="col-name">${r.proyek}</td>
            <td><span class="k-badge">${r.kontrak}</span></td>
            <td class="col-mono">${r.lopid}</td>
            <td>${r.pelanggan}</td>
            <td><span class="s-badge ${statusClass(r.status)}">${r.status}</span></td>
            <td class="col-mono col-right">${fmtNilai(r.nilai)}</td>
            <td style="font-size:11px;color:var(--text-2)">${r.est}</td>
          </tr>
        `);
      });
    });
  });

  if (!html.length) {
    html.push(`<tr class="empty-row"><td colspan="7">Tidak ada data yang cocok dengan filter aktif</td></tr>`);
  }

  document.getElementById('tbl-body').innerHTML = html.join('');
  document.getElementById('lop-count-badge').textContent = totalLOP;
}

function toggleExpandAll() {
  state.allExpanded = !state.allExpanded;
  const data = getFilteredData();
  const amNames = [...new Set(data.map(r=>r.am))];
  const phases = ['F0','F1','F2','F3','F4','F5'];
  amNames.forEach(am => {
    state.expandedAM[am] = state.allExpanded;
    phases.forEach(p => {
      state.expandedPhase[`${am}|${p}`] = state.allExpanded;
    });
  });
  document.getElementById('expand-all-btn').textContent = state.allExpanded ? 'Collapse semua' : 'Expand semua';
  render();
}

function toggleDd(id) {
  document.querySelectorAll('.dropdown-menu').forEach(m => {
    if (m.id !== id) { m.classList.remove('open'); }
  });
  document.querySelectorAll('.dropdown-trigger').forEach(t => {
    if (t.id !== 'trig-' + id.replace('dd-','')) t.classList.remove('open');
  });
  const menu = document.getElementById(id);
  const trigId = 'trig-' + id.replace('dd-','');
  const trig = document.getElementById(trigId);
  menu.classList.toggle('open');
  if (trig) trig.classList.toggle('open', menu.classList.contains('open'));
}
function closeDd(id) {
  document.getElementById(id).classList.remove('open');
  const trigId = 'trig-' + id.replace('dd-','');
  const trig = document.getElementById(trigId);
  if (trig) trig.classList.remove('open');
  render();
}

// Close on outside click
document.addEventListener('click', e => {
  if (!e.target.closest('.dropdown-wrapper')) {
    document.querySelectorAll('.dropdown-menu').forEach(m => m.classList.remove('open'));
    document.querySelectorAll('.dropdown-trigger').forEach(t => t.classList.remove('open'));
  }
});

function toggleAM(val) {
  if (val === 'all') {
    const allChk = document.getElementById('chk-am-all');
    const checked = !allChk.checked;
    allChk.checked = checked;
    document.querySelectorAll('.am-chk').forEach(c => c.checked = checked);
  } else {
    const chk = document.querySelector(`.am-chk[data-am="${val}"]`);
    if (chk) chk.checked = !chk.checked;
    const allChecked = [...document.querySelectorAll('.am-chk')].every(c=>c.checked);
    document.getElementById('chk-am-all').checked = allChecked;
  }
  updateAMState();
}
function updateAMState() {
  const all = [...document.querySelectorAll('.am-chk')];
  const checked = all.filter(c=>c.checked).map(c=>c.dataset.am);
  state.am = checked.length === all.length ? null : (checked.length ? checked : null);
  const lbl = state.am ? (state.am.length === 1 ? state.am[0].split(' ')[0] : `${state.am.length} AM`) : 'Semua AM';
  document.getElementById('am-label').innerHTML = lbl + (state.am ? `<span class="active-badge">${state.am.length}</span>` : '');
}
function clearAM() {
  document.querySelectorAll('.am-chk, #chk-am-all').forEach(c=>c.checked=true);
  state.am = null;
  document.getElementById('am-label').textContent = 'Semua AM';
  render();
}
