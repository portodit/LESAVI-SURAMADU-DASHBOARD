function toggleDD(key) {
  const panel   = document.getElementById(`dd-${key}-panel`);
  const trigger = document.getElementById(`dd-${key}-trigger`);
  if (ddOpen && ddOpen !== key) closeDD(ddOpen);

  }

  document.getElementById(`dd-${key}-panel`)?.classList.remove('open');

function getChecked(key) {

function updateDDLabel(key) {
  const all     = [...document.querySelectorAll(`#dd-${key}-list input[type=checkbox]`)];
  const checked = all.filter(c=>c.checked);
  const label   = document.getElementById(`dd-${key}-label`);
  const count   = document.getElementById(`dd-${key}-count`);
  if (checked.length === 0) {
    label.textContent = 'Tidak ada';
    count.style.display = 'none';
  } else if (checked.length === all.length) {
    label.textContent = key === 'am' ? 'Semua AM' : 'Semua Kategori';
    count.style.display = 'none';
  } else {
    if (key === 'am') {
      label.textContent = checked.length === 1 ? checked[0].value.split(' ')[0] : `${checked.length} AM dipilih`;
    } else {
      label.textContent = `${checked.length} Kategori`;
    }
    count.textContent     = checked.length;
    count.style.display   = 'inline';
  }
}

function getFiltered() {
  const bulan  = document.getElementById('f-bulan').value;
  const tahun  = parseInt(document.getElementById('f-tahun').value);
  const divisi = document.getElementById('f-divisi').value;
  const selAMs = new Set(getChecked('am'));
  const selKat = new Set(getChecked('kat'));
  const search = (document.getElementById('search-am')?.value||'').toLowerCase().trim();

  return RAW.filter(d => {
    const dd = new Date(d.date);
    if (dd.getFullYear() !== tahun) return false;
    if (bulan !== 'all' && dd.getMonth()+1 !== parseInt(bulan)) return false;
    if (divisi !== 'all' && d.div !== divisi) return false;
    if (!selAMs.has(d.am)) return false;
    if (search && !d.am.toLowerCase().includes(search) && !d.cust.toLowerCase().includes(search)) return false;
    return true;
  }).map(d => ({...d, isKPI: selKat.has(d.cat)}));
}

function applyFilters() {
  const data = getFiltered();
  renderOverview(data);
  renderTable(data);
}

function renderOverview(data) {
  const kpi  = data.filter(d=>d.isKPI);
  const ams  = [...new Set(data.map(d=>d.am))];
  const byAM = groupBy(kpi,'am');
  const reach = ams.filter(am=>(byAM[am]?.length||0)>=KPI).length;
  const below = ams.length - reach;

  animNum('ov-total', kpi.length);
  animNum('ov-reach', reach);
  animNum('ov-below', below);
  document.getElementById('ov-total-am').textContent = ams.length;
  document.getElementById('ov-below-sub').textContent =
    below === 0 ? 'Semua AM mencapai target 🎉' : `${below} AM perlu perhatian lebih`;
}
function animNum(id, target) {
  const el = document.getElementById(id);
  const from = parseInt(el.textContent)||0;
  const dur=450; let t=0;
  const timer = setInterval(()=>{
    t+=16; const p=Math.min(t/dur,1); const e=1-Math.pow(1-p,3);
    el.textContent=Math.round(from+(target-from)*e);
    if(p>=1) clearInterval(timer);
  },16);
}

function toggleRow(am, rowId) {
  const row = document.getElementById(rowId);
  if (!row) return;
  if (expandedRows.has(am)) {
    expandedRows.delete(am);
    row.classList.remove('expanded');
    row.querySelector('.tog').textContent = '+';
  } else {
    expandedRows.add(am);
    row.classList.add('expanded');
    row.querySelector('.tog').textContent = '−';
  }
}
function expandAll() {
  document.querySelectorAll('.am-row').forEach(row=>{
    row.classList.add('expanded');
    row.querySelector('.tog').textContent='−';
    const am = row.querySelector('.am-name-text').title;
    expandedRows.add(am);
  });
}
function collapseAll() {
  document.querySelectorAll('.am-row').forEach(row=>{
    row.classList.remove('expanded');
    row.querySelector('.tog').textContent='+';
  });
  expandedRows.clear();
}

function groupBy(arr,key){
  return arr.reduce((a,c)=>{(a[c[key]]=a[c[key]]||[]).push(c);return a;},{});
}
